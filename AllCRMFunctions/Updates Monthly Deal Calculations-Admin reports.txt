	thisMonthWonDeal = ifnull("Deals_Count_Monthly",0);
	lastMonthWonAmount = ifnull("Won_Amount_Last_Month",0);
	lastMonthWonDeal = ifnull("Deals_Count_Last_Month",0);

}
}
