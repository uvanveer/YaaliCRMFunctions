		if(tagsExist == true)
		{
			tagsData = highRiseContactsInfo.executexpath("person/tags/tag").toXmlList();
			tags = list();
			for each  index_item in tagsData
			{
				tagId = index_item.executexpath("tag/id/text()");
				tagText = index_item.executexpath("tag/name/text()");
				insertTag = Map();
				insertTag.put("Contact",zohoCrmContactId);
				insertTag.put("Name",tagText);
				insertTag.put("Highrise_Tag_Id",tagId);
				insertTag.put("Highrise_Module_Name","Contacts");
				logInfo = zoho.crm.createRecord("Highrise_Tags",insertTag);
				info logInfo;
				tags.add(logInfo.get("id"));
				// Updated contact 
			}
			// update Zoho CRM contact
			info tags;
			updateContact = Map();
			updateContact.put("Highrise_Tags1",tags);
			updatedInfo = zoho.crm.updateRecord("Contacts",zohoCrmContactId,updateContact);
			info updatedInfo;
		}
	}
