string button.Change_Deal_Status_To_Pending(String Deal_Id)
{
info Deal_Id;
Deal = zoho.crm.getRecordById("Deals",Deal_Id);
info Deal.get("Stage");
if(Deal.get("Stage") != "WON")
{
	Deal_Update = zoho.crm.updateRecord("Deals",Deal_Id,{"Stage":"PENDING"});
	info Deal_Update;
	return "Status has been changed to 'PENDING' Successfully " + Deal_Id;
}
else
{
	return "Status has not been changed to 'PENDING' Successfully " + Deal_Id;
}
}
