void automation.hide_section_in_contact(Int contact_id)
{
contact = zoho.crm.getRecordById("Contacts",contact_id);
userTypeMap = Map();
userTypeMap.put("type","ActiveUsers");
response = zoho.crm.invokeConnector("crm.getusers",userTypeMap);
user_list = response.get("response").get("users").toJSONList();
for each  user in user_list
{
	user_email = user.get("email");
	user_profile = user.get("profile").get("name");
	if(user_email == zoho.loginuserid)
	{
		main_map = Map();
		owner_map = Map();
		user_id = user.get("id");
		user_name = user.get("full_name");
		owner_map.put("id",user_id);
		owner_map.put("name",user_name);
		main_map.put("Owner",owner_map);
		if(user_profile == "Sales with Chat")
		{
			owner_update = zoho.crm.updateRecord("Contacts",contact_id,main_map);
		}
		else
		{
			main_map.put("view_status",true);
			owner_update = zoho.crm.updateRecord("Contacts",contact_id,main_map);
			info owner_update;
		}
		break;
	}
}
}
